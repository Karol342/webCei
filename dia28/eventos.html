Mouse events

Click - el más usado

element.addEventListener('click', function(e) { console.log('Elemento clickeado'); });

Hover effects

element.addEventListener('mouseenter', function(e) {});

element.addEventListener('mouseleave', function(e) {});

Mouse sobre elemento

element.addEventListener('mouseover', function(e) {});

Form events

Submit - envío de formularios

form.addEventListener('submit', function(e) { e.preventDefault()});

Input - cambios en tiempo real

input.addEventListener('input', function(e) { console.log('Valor actual:', e.target.value);});

Change

select.addEventListener('change', function(e) { console.log('Opción seleccionada:', e.target.value);});

Focus y Blur

input.addEventListener('focus', function(e) {});

input.addEventListener('blur', function(e) {});

Keyboard events

Keydown - tecla presionada

document.addEventListener('keydown', function(e) { if (e.key === 'Enter') { console.log('Enter presionado'); } if (e.key === 'Escape') {} }); input.addEventListener('keyup', function(e) {});

DOM Event

DOMContentLoaded

document.addEventListener('DOMContentLoaded', function() { console.log('DOM listo'); });

Load

window.addEventListener('load', function() { });

Resize

window.addEventListener('resize', function() { console.log('Ventana redimensionada'); });

Scroll

window.addEventListener('scroll', function() { console.log('Scroll position:', window.scrollY); });

Even Touch

element.addEventListener('touchstart', function(e) { });// Toque iniciado

element.addEventListener('touchend', function(e) { });// Toque terminado

element.addEventListener('touchmove', function(e) { e.preventDefault(); // Prevenir scroll}); // Deslizamiento
